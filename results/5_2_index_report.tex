\section{Local Charges Formula}
The refined index is computed by the formula:
\begin{equation}
  I^{\text{ref}}(q;\, \eta_0, \ldots, \eta_{k-1})
  = \sum_{e_{\text{int}} \in (\frac{1}{2}\mathbb{Z})^{n-r}}
    \biggl[\prod_{a=0}^{k-1} \eta_a^{e_{r+a}}\biggr]
    \cdot (-q^{1/2})^{\varphi}
    \cdot \prod_{j=1}^{n}
      I_\Delta\bigl((g_{NZ}^{-1}\kappa)_j,\;
                     (g_{NZ}^{-1}\kappa)_{n+j}\bigr)
\end{equation}
where:
\begin{itemize}
  \item $\kappa = (m_{\text{ext}},\, \underbrace{0,\ldots,0}_{n-r},\; e_{\text{ext}},\, e_{\text{int}})$ is the $2n$-vector of external + internal charges.
  \item The \textbf{local charges} (tetrahedron arguments) are $(m_j, e_j) = (g_{NZ}^{-1}\kappa)_j$ and $(g_{NZ}^{-1}\kappa)_{n+j}$.
  \item The \textbf{phase exponent} is $\varphi = m_{\text{full}} \cdot \nu_p - e_{\text{full}} \cdot \nu_x$.
  \item $I_\Delta(m, e)$ is the tetrahedron index.
  \item $k = \text{num\_hard}$ is the number of hard edges carrying fugacity variables.
\end{itemize}
